%%%%%%%%%%%%%%%%%%%%%%%%%%%%% Fußnoten
\DeclareCiteCommand{\bookcite}[\mkbibfootnote]
    {\printtext{Vgl.}\space
    }
    {\usebibmacro{citeindex}%
        {\toggletrue{abx@bool@giveninits}
        \textit{\printnames[family-given]{labelname}}}%
        \addcomma\space\printfield{shorttitle}
        \addcomma\space\printfield{year}
    }
    {\addcomma\space\printtext{S.}\space}
    {\usebibmacro{postnote}
    }

\DeclareCiteCommand{\directcite}[\mkbibfootnote]
    {}
    {\usebibmacro{citeindex}%
        {\toggletrue{abx@bool@giveninits}
        \textit{\printnames[family-given]{labelname}}}%
        \addcomma\space\printfield{shorttitle}
        \addcomma\space\printfield{year}
    }
    {\addsemicolon\space}
    {\usebibmacro{postnote}
    }

\DeclareCiteCommand{\articlecite}[\mkbibfootnote]
    {\printtext{Vgl.}\space
    }
    {\usebibmacro{citeindex}%
        \textit{\printnames[sortname][1-1]{author}}%
        \addcomma\space\printfield{shorttitle}
        \addcomma\space\printfield{year}
    }
    {\addcomma\space\printtext{S.}\space}
    {\usebibmacro{postnote}
    }

\DeclareCiteCommand{\onlinecite}[\mkbibfootnote]
    {\printtext{Vgl.}}                                        
    {\usebibmacro{citeindex}%                                 
        %\space\textit{\printfield{url}}
        \addcomma\space\printfield{shorttitle}
        \addcomma\space
        \space\printfield{usera}
        %\space\printfield{time}
    }
    {}
    {\usebibmacro{postnote}
    }
